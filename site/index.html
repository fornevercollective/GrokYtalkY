Install

Companion dock + mesh hub

git clone https://github.com/fornevercollective/GrokYtalkY.git
cd GrokYtalkY
make install                  # → ~/.local/bin/gy
# or: gy install · gy update · gy upgrade
#     gy clean install · gy uninstall
#     gy install dependencies [--yes]
gy                            # companion dock (auto-update)
gy serve                      # hub + site/ (GrokGlyph PWA)
gy watch clip.mp4
